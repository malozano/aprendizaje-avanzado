    from sklearn.svm import SVC
from sklearn.model_selection import cross_val_score
from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import Pipeline

# CRÍTICO: SVM requiere escalado
pipe_linear = Pipeline([
    ('scaler', StandardScaler()),
    ('svm', SVC(kernel='linear', random_state=42))
])

pipe_poly = Pipeline([
    ('scaler', StandardScaler()),
    ('svm', SVC(kernel='poly', degree=3, random_state=42))
])

pipe_rbf = Pipeline([
    ('scaler', StandardScaler()),
    ('svm', SVC(kernel='rbf', random_state=42))
])

pipe_sigmoid = Pipeline([
    ('scaler', StandardScaler()),
    ('svm', SVC(kernel='sigmoid', random_state=42))
])

# Entrenar y evaluar cada uno
for name, pipe in [('Linear', pipe_linear), ('Poly', pipe_poly), 
                   ('RBF', pipe_rbf), ('Sigmoid', pipe_sigmoid)]:
    pipe.fit(X_train, y_train)
    scores = cross_val_score(pipe, X_train, y_train, cv=5, scoring='accuracy')
    print(f"{name}: {scores.mean():.4f} (+/- {scores.std():.4f})")

    # Obtener vectores de soporte
    n_sv = pipe.named_steps['svm'].n_support_
    print(f"Vectores de soporte: {n_sv.sum()} ({n_sv.sum()/len(X_train)*100:.1f}% del training)")

    2. Optimización del Kernel RBF¶

    El kernel RBF suele ser el más efectivo. Explora el efecto de sus hiperparámetros C y gamma.

    Hiperparámetro C: - Controla el trade-off entre maximizar el margen y minimizar errores de clasificación - C pequeño → margen grande, más errores permitidos (puede causar underfitting) - C grande → margen pequeño, menos errores (puede causar overfitting)

    Hiperparámetro gamma: - Controla la influencia de cada sample en la frontera de decisión - gamma pequeño → influencia amplia, frontera suave - gamma grande → influencia local, frontera compleja (puede causar overfitting)

    from sklearn.model_selection import GridSearchCV

pipe_rbf = Pipeline([
    ('scaler', StandardScaler()),
    ('svm', SVC(kernel='rbf', random_state=42))
])

param_grid = {
    'svm__C': [0.1, 1, 10, 100],
    'svm__gamma': [0.001, 0.01, 0.1, 1, 'scale', 'auto']
}

grid_search = GridSearchCV(
    pipe_rbf,
    param_grid,
    cv=5,
    scoring='accuracy',
    n_jobs=-1,
    verbose=1
)

grid_search.fit(X_train, y_train)

print(f"Mejores parámetros: {grid_search.best_params_}")
print(f"Mejor score CV: {grid_search.best_score_:.4f}")

# Evaluar en test
best_model = grid_search.best_estimator_
y_pred = best_model.predict(X_test)
print(f"Test accuracy: {accuracy_score(y_test, y_pred):.4f}")

    3. Optimización del Kernel Polinomial¶

    El kernel polinomial puede capturar relaciones polinomiales sin transformar explícitamente las features.

    4. Comparación de Implementaciones (Kernel Lineal)¶

    Scikit-learn ofrece tres implementaciones diferentes para SVM lineal. Compara su rendimiento y eficiencia.

    SVC vs LinearSVC vs SGDClassifier:

    Implementación Formulación Cuándo usar
    SVC(kernel='linear') Dual (libsvm) n_samples < 10,000
    LinearSVC Primal (liblinear) Kernel lineal, datasets medianos
    SGDClassifier(loss='hinge') SGD Datasets grandes (> 100,000) 
    from sklearn.svm import SVC, LinearSVC
from sklearn.linear_model import SGDClassifier
from time import time

models = {
    'SVC(linear)': Pipeline([
        ('scaler', StandardScaler()),
        ('svm', SVC(kernel='linear', random_state=42))
    ]),
    'LinearSVC': Pipeline([
        ('scaler', StandardScaler()),
        ('svm', LinearSVC(random_state=42, max_iter=10000))
    ]),
    'SGDClassifier(hinge)': Pipeline([
        ('scaler', StandardScaler()),
        ('svm', SGDClassifier(loss='hinge', random_state=42))
    ])
}

results = {}
for name, model in models.items():
    start = time()
    model.fit(X_train, y_train)
    train_time = time() - start

    test_acc = model.score(X_test, y_test)
    cv_scores = cross_val_score(model, X_train, y_train, cv=5)

    results[name] = {
        'CV Accuracy': cv_scores.mean(),
        'Test Accuracy': test_acc,
        'Train Time': train_time
    }

    # Vectores de soporte (solo SVC los tiene)
    if 'SVC' in name and hasattr(model.named_steps['svm'], 'n_support_'):
        n_sv = model.named_steps['svm'].n_support_.sum()
        results[name]['N° SV'] = n_sv

    5. Análisis de Vectores de Soporte¶

    Los vectores de soporte son los puntos críticos que definen la frontera de decisión.

    ¿Qué indican? - Pocos vectores de soporte (< 20%) → modelo simple, buena generalización probable - Muchos vectores de soporte (> 50%) → modelo complejo, posible overfitting
